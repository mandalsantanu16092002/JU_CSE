void iterationtest(){
    srand(time(NULL));
    for(int i=0; i<10; i++)
    {
            printf("\nPID %d: Iteration no %d\n", getpid(), i+1);
            sleep(rand()%5);
    }
}

int main()
{
    int fd[2];
    if(pipe(fd)<0) {return 2;}
    
    int id = fork();
    if(id<0) {return 1;}
    if(id!=0){
        //Process X
        iterationtest();
        wait(NULL);
    }
    else{
        //Process Y
        iterationtest();
    }
    return 0;
}
